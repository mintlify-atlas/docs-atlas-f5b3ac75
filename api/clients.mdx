## Overview

The Clients API provides CRUD operations for managing hotel guest information. All endpoints are defined in `router/clientes.Router.js:1-12` and implemented in `controllers/clientes.controllers.js:1-38`.

<Info>
  Base URL: `http://localhost:4000`
</Info>

## Endpoints Summary

| Method | Endpoint | Description |
|--------|----------|-------------|
| GET | `/clientes` | Get all clients |
| POST | `/clientes` | Create a new client |
| PUT | `/clientes/:id` | Update a client |
| DELETE | `/clientes/:id` | Delete a client |

## Database Schema

The `cliente` table has the following structure:

| Field | Type | Constraints |
|-------|------|-------------|
| `id` | INT | PRIMARY KEY, AUTO_INCREMENT |
| `nombre` | VARCHAR | NOT NULL |
| `correo` | VARCHAR | NOT NULL |
| `telefono` | VARCHAR | NOT NULL |

## Complete Workflow Example

Here's a complete example showing how to create, read, update, and delete a client:

```javascript Complete CRUD Example
const API_URL = 'http://localhost:4000/clientes';

// 1. Create a new client
async function createClient() {
  const response = await fetch(API_URL, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      nombre: 'Alice Johnson',
      correo: 'alice@example.com',
      telefono: '555-0147'
    })
  });
  const data = await response.json();
  console.log('Created:', data);
}

// 2. Get all clients
async function getAllClients() {
  const response = await fetch(API_URL);
  const clients = await response.json();
  console.log('All clients:', clients);
  return clients;
}

// 3. Update a client
async function updateClient(id) {
  const response = await fetch(`${API_URL}/${id}`, {
    method: 'PUT',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      nombre: 'Alice Johnson-Smith',
      correo: 'alice.smith@example.com',
      telefono: '555-0147'
    })
  });
  const data = await response.json();
  console.log('Updated:', data);
}

// 4. Delete a client
async function deleteClient(id) {
  const response = await fetch(`${API_URL}/${id}`, {
    method: 'DELETE'
  });
  const data = await response.json();
  console.log('Deleted:', data);
}

// Run the workflow
(async () => {
  await createClient();
  const clients = await getAllClients();
  const newClientId = clients[clients.length - 1].id;
  await updateClient(newClientId);
  await deleteClient(newClientId);
})();
```

## Error Handling

The current implementation throws errors on database failures. For production, implement proper error handling:

```javascript Enhanced Error Handling
exports.obtenerListClientesService = (req, res) => {
    const SQL = "SELECT * FROM `cliente`";
    conexion.query(SQL, (err, results) => {
        if (err) {
            console.error('Database error:', err);
            return res.status(500).json({ 
                message: 'Error al obtener clientes',
                error: process.env.NODE_ENV === 'development' ? err.message : undefined
            });
        }
        res.json(results);
    });
};
```

## Validation Best Practices

Add input validation to ensure data quality:

```javascript Input Validation
const { body, validationResult } = require('express-validator');

router.post('/clientes', [
  body('nombre')
    .trim()
    .isLength({ min: 2, max: 100 })
    .withMessage('Nombre debe tener entre 2 y 100 caracteres'),
  body('correo')
    .isEmail()
    .normalizeEmail()
    .withMessage('Correo inválido'),
  body('telefono')
    .matches(/^[0-9-+() ]+$/)
    .withMessage('Teléfono inválido')
], (req, res) => {
  const errors = validationResult(req);
  if (!errors.isEmpty()) {
    return res.status(400).json({ errors: errors.array() });
  }
  service.addDataCliente(req, res);
});
```
