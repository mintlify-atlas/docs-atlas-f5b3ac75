## Overview

The Rooms API provides CRUD operations for managing hotel room inventory. All endpoints are defined in `router/habitaciones.Router.js:1-12` and implemented in `controllers/habitaciones.controllers.js:1-38`.

<Info>
  Base URL: `http://localhost:4000`
</Info>

## Endpoints Summary

| Method | Endpoint | Description |
|--------|----------|-------------|
| GET | `/habitaciones` | Get all rooms |
| POST | `/habitaciones` | Create a new room |
| PUT | `/habitaciones/:id` | Update a room |
| DELETE | `/habitaciones/:id` | Delete a room |

## Database Schema

The `habitaciones` table has the following structure:

| Field | Type | Constraints |
|-------|------|-------------|
| `id` | INT | PRIMARY KEY, AUTO_INCREMENT |
| `nombre` | VARCHAR | NOT NULL |
| `tipo` | VARCHAR | NOT NULL |
| `precio` | DECIMAL | NOT NULL |

## Room Types Reference

Common room types used in the system:

  <Accordion title="Individual">
    Single occupancy room with one bed. Ideal for solo travelers.
    
    **Typical pricing**: $40 - $60 per night
  </Accordion>
  
  <Accordion title="Doble">
    Double occupancy room with two beds or one double bed. Suitable for couples or two guests.

  <Accordion title="Suite">

## Complete Room Management Example

```javascript Room Management Workflow
const API_URL = 'http://localhost:4000/habitaciones';

// Add multiple rooms to inventory
async function initializeRoomInventory() {
  const rooms = [
    { nombre: '101', tipo: 'Individual', precio: 50.00 },
    { nombre: '102', tipo: 'Individual', precio: 50.00 },
    { nombre: '201', tipo: 'Doble', precio: 75.00 },
    { nombre: '202', tipo: 'Doble', precio: 75.00 },
    { nombre: '301', tipo: 'Suite', precio: 200.00 }
  ];
  
  for (const room of rooms) {
    await fetch(API_URL, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(room)
    });
  }
  
  console.log('Room inventory initialized');
}

// Get available rooms by type
async function getRoomsByType(tipo) {
  const response = await fetch(API_URL);
  const rooms = await response.json();
  return rooms.filter(room => room.tipo === tipo);
}

// Update room pricing (e.g., seasonal adjustment)
async function updateRoomPricing(roomId, newPrice) {
  // First, get current room data
  const response = await fetch(API_URL);
  const rooms = await response.json();
  const room = rooms.find(r => r.id === roomId);
  
  if (!room) {
    console.error('Room not found');
    return;
  }
  
  // Update with new price
  await fetch(`${API_URL}/${roomId}`, {
    method: 'PUT',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      ...room,
      precio: newPrice
    })
  });
  
  console.log(`Room ${roomId} price updated to $${newPrice}`);
}

// Usage
(async () => {
  await initializeRoomInventory();
  const suites = await getRoomsByType('Suite');
  console.log('Available suites:', suites);
  await updateRoomPricing(1, 55.00); // Increase price
})();
```

## Validation Recommendations

```javascript Input Validation
const { body, validationResult } = require('express-validator');

router.post('/habitaciones', [
  body('nombre')
    .trim()
    .notEmpty()
    .withMessage('Nombre de habitación requerido'),
  body('tipo')
    .isIn(['Individual', 'Doble', 'Suite', 'Familiar'])
    .withMessage('Tipo de habitación inválido'),
  body('precio')
    .isFloat({ min: 0 })
    .withMessage('Precio debe ser un número positivo')
], (req, res) => {
  const errors = validationResult(req);
  if (!errors.isEmpty()) {
    return res.status(400).json({ errors: errors.array() });
  }
  service.addDataHabitaciones(req, res);
});
```
