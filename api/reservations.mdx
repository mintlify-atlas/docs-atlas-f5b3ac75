## Overview

The Reservations API provides CRUD operations for managing hotel bookings. All endpoints are defined in `router/reservaciones.Router.js:1-12` and implemented in `controllers/reservaciones.controllers.js:1-55`.

<Info>
  Base URL: `http://localhost:4000`
</Info>

## Endpoints Summary

| Method | Endpoint | Description |
|--------|----------|-------------|
| GET | `/reservaciones` | Get all reservations |
| POST | `/reservaciones` | Create a new reservation |
| PUT | `/reservaciones/:id` | Update a reservation |
| DELETE | `/reservaciones/:id` | Delete a reservation |

## Database Schema

The `reservaciones` table has the following structure:

| Field | Type | Constraints |
|-------|------|-------------|
| `id` | INT | PRIMARY KEY, AUTO_INCREMENT |
| `folio` | VARCHAR | NOT NULL |
| `nombrecliente` | VARCHAR | NOT NULL |

<Note>
  The current schema is simplified. For production, consider adding fields like `fecha_inicio`, `fecha_fin`, `habitacion_id`, `cliente_id`, and `estado`.
</Note>

## Folio Generation

Folio numbers are unique identifiers for reservations. Here's a recommended pattern:

```javascript Folio Generator
function generateFolio() {
  const date = new Date();
  const year = date.getFullYear();
  const month = String(date.getMonth() + 1).padStart(2, '0');
  const day = String(date.getDate()).padStart(2, '0');
  const timestamp = Date.now().toString().slice(-4);
  
  return `RES-${year}${month}${day}-${timestamp}`;
}

// Example output: RES-20240315-7892
const folio = generateFolio();
console.log(folio);
```

## Complete Reservation Workflow

```javascript Reservation Management
const API_URL = 'http://localhost:4000/reservaciones';

// Generate unique folio
function generateFolio() {
  const timestamp = Date.now();
  const random = Math.floor(Math.random() * 1000);
  return `RES-${timestamp}-${random}`;
}

// Create a new reservation
async function createReservation(clientName) {
  const folio = generateFolio();
  
  const response = await fetch(API_URL, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      folio: folio,
      nombrecliente: clientName
    })
  });
  
  const data = await response.json();
  console.log('Reservation created:', data);
  return folio;
}

// Find reservation by folio
async function findReservationByFolio(folio) {
  const response = await fetch(API_URL);
  const reservations = await response.json();
  return reservations.find(r => r.folio === folio);
}

// Update reservation client name
async function updateReservationClient(reservationId, newClientName) {
  // Get current reservation
  const response = await fetch(API_URL);
  const reservations = await response.json();
  const reservation = reservations.find(r => r.id === reservationId);
  
  if (!reservation) {
    console.error('Reservation not found');
    return;
  }
  
  // Update with new client name
  await fetch(`${API_URL}/${reservationId}`, {
    method: 'PUT',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      folio: reservation.folio,
      nombrecliente: newClientName
    })
  });
  
  console.log('Reservation updated');
}

// Cancel reservation
async function cancelReservation(reservationId) {
  const response = await fetch(`${API_URL}/${reservationId}`, {
    method: 'DELETE'
  });
  
  const data = await response.json();
  console.log('Reservation cancelled:', data);
}

// Usage
(async () => {
  const folio = await createReservation('Alice Johnson');
  const reservation = await findReservationByFolio(folio);
  console.log('Found reservation:', reservation);
  
  if (reservation) {
    await updateReservationClient(reservation.id, 'Alice Johnson-Smith');
    // await cancelReservation(reservation.id);
  }
})();
```

## Enhanced Schema Recommendation

For a production-ready reservation system, consider this enhanced schema:

```sql Enhanced Schema
CREATE TABLE `reservaciones` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `folio` VARCHAR(50) UNIQUE NOT NULL,
  `cliente_id` INT,
  `habitacion_id` INT,
  `fecha_inicio` DATE NOT NULL,
  `fecha_fin` DATE NOT NULL,
  `numero_huespedes` INT DEFAULT 1,
  `estado` ENUM('pendiente', 'confirmada', 'cancelada', 'completada') DEFAULT 'pendiente',
  `precio_total` DECIMAL(10, 2),
  `notas` TEXT,
  `creado_en` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  `actualizado_en` TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (`cliente_id`) REFERENCES `cliente`(`id`),
  FOREIGN KEY (`habitacion_id`) REFERENCES `habitaciones`(`id`)
);
```

With this schema, you could implement:
- Date range validation
- Room availability checking
- Automatic price calculation
- Reservation status tracking
- Guest count management

## Integration Example

Combine clients, rooms, and reservations:

```javascript Full Booking Flow
class HotelBookingSystem {
  constructor(baseURL) {
    this.baseURL = baseURL;
  }
  
  async createCompleteReservation(clientData, roomType) {
    // 1. Create or find client
    const client = await this.createClient(clientData);
    
    // 2. Find available room
    const room = await this.findAvailableRoom(roomType);
    
    if (!room) {
      throw new Error('No rooms available of type: ' + roomType);
    }
    
    // 3. Create reservation
    const folio = this.generateFolio();
    const reservation = await this.createReservation({
      folio: folio,
      nombrecliente: client.nombre
    });
    
    return {
      client,
      room,
      reservation,
      folio
    };
  }
  
  async createClient(data) {
    const response = await fetch(`${this.baseURL}/clientes`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(data)
    });
    return response.json();
  }
  
  async findAvailableRoom(tipo) {
    const response = await fetch(`${this.baseURL}/habitaciones`);
    const rooms = await response.json();
    return rooms.find(r => r.tipo === tipo);
  }
  
  async createReservation(data) {
    const response = await fetch(`${this.baseURL}/reservaciones`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(data)
    });
    return response.json();
  }
  
  generateFolio() {
    return `RES-${Date.now()}-${Math.floor(Math.random() * 1000)}`;
  }
}

// Usage
const hotel = new HotelBookingSystem('http://localhost:4000');

hotel.createCompleteReservation(
  {
    nombre: 'Carlos Mendez',
    correo: 'carlos@example.com',
    telefono: '555-0177'
  },
  'Suite'
).then(booking => {
  console.log('Booking complete:', booking);
  console.log('Folio:', booking.folio);
});
```
