## Overview

The Client Management module allows you to create, read, update, and delete client information in the Hotel Reservation System. This guide covers all client operations with real code examples from the application.

<Note>
  Client data includes name, email, and phone number. All operations use the REST API endpoint `http://localhost:4000/clientes`.
</Note>

## Client Data Structure

Each client record contains the following fields:

| Field | Type | Description |
|-------|------|-------------|
| `id` | Integer | Auto-generated unique identifier |
| `nombre` | String | Client's full name |
| `correo` | String | Client's email address |
| `telefono` | String | Client's phone number |

## Getting Started

<Steps>

### Step 1: Access the Client Management Page

Navigate to the client management interface at `index.html`. The page displays a table of all registered clients with action buttons for each entry.

## CRUD Operations

### Reading Client Data

The application automatically loads all clients when the page loads using the `loadDataCliente()` function.

#### Loading All Clients

```javascript
async function loadDataCliente() {
    try {
        const response = await fetch(URL);
        data = await response.json();

        let html = '';
        data.forEach(cliente => {
            html += `
            <tr>
                <td>${cliente.id}</td>
                <td>${cliente.nombre}</td>
                <td>${cliente.correo}</td>
                <td>${cliente.telefono}</td>
                <td>
                    <button type="button" class="btn btn-primary btn-sm" onclick="modificarDataCliente(${cliente.id})">Actualizar</button>
                    <button type="button" class="btn btn-danger btn-sm" onclick="deleteDataCliente(${cliente.id})">Eliminar</button>
                </td>
            </tr>`;
        });

        T_DATA_CLIENTE.innerHTML = html;

    } catch (error) {
        console.error(error);
    }
}
```

**Key Points:**
- Uses `fetch()` with GET method (default)
- Parses JSON response
- Dynamically generates table rows
- Includes action buttons with client ID

#### Backend Controller

The backend retrieves clients from the MySQL database:

```javascript
exports.obtenerListClientesService = (req, res) => {
    const SQL = "SELECT * FROM `cliente`";
    conexion.query(SQL, (err, results) => {
        if (err) throw err;
        res.json(results);
    });

```

<Note>
  The function executes a SELECT query and returns all client records as JSON.
</Note>

### Updating Existing Clients

Updating a client is a two-step process: opening the update dialog and submitting the changes.

#### Opening the Update Dialog

```javascript
function modificarDataCliente(id) {
    const dataCliente = data.find(cliente => cliente.id === id);
    document.getElementById("clientNameU").value = dataCliente.nombre;
    document.getElementById("clientEmailU").value = dataCliente.correo;
    document.getElementById("clientPhoneU").value = dataCliente.telefono;

    const dialog = document.getElementById("formDataCliente");
    dialog.dataset.clienteId = id;
    dialog.showModal();
}
```

**Key Points:**
- Finds the client by ID in the local data array
- Pre-fills the update form with current values
- Stores the client ID in the dialog's dataset
- Opens the native HTML `<dialog>` element

#### Submitting Updates

```javascript
async function actualizarDataCliente() {
    const dialog = document.getElementById("formDataCliente");
    const id = dialog.dataset.clienteId;

    const dataCliente = {
        nombre: document.getElementById("clientNameU").value,
        correo: document.getElementById("clientEmailU").value,
        telefono: document.getElementById("clientPhoneU").value
    };

    try {
        const response = await fetch(`${URL}/${id}`, {
            method: "PUT",
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(dataCliente)
        });

        if (!response.ok) throw new Error('Error al actualizar cliente');

        dialog.close();
        loadDataCliente();

    } catch (error) {
        console.error(error);
    }
}
```

<Warning>
  Always validate that the client ID exists before sending an update request to avoid errors.
</Warning>

#### Backend Update Handler

```javascript
exports.updateDataCliente = (req, res) => {
    const id = req.params.id;
    const { nombre, correo, telefono } = req.body;
    const SQL = "UPDATE `cliente` SET nombre=?, correo=?, telefono=? WHERE id=?";
    conexion.query(SQL, [nombre, correo, telefono, id], (err, results) => {
        if (err) throw err;
        res.json({ message: "Cliente actualizado" });
    });
};
```

**API Route:** `PUT /clientes/:id`

### Deleting Clients

Deleting a client removes the record permanently from the database.

```javascript
async function deleteDataCliente(id) {
    try {
        const response = await fetch(`${URL}/${id}`, { method: "DELETE" });
        if (!response.ok) throw new Error('Error al eliminar cliente');

        loadDataCliente();

    } catch (error) {
        console.error(error);
    }
}
```

<Warning>
  This operation is irreversible. Consider adding a confirmation dialog before deleting clients in production.
</Warning>

#### Backend Delete Handler

```javascript
exports.deleteDataCliente = (req, res) => {
    const id = req.params.id;
    const SQL = "DELETE FROM `cliente` WHERE id=?";
    conexion.query(SQL, id, (err, results) => {
        if (err) throw err;
        res.json({ message: "Cliente eliminado" });
    });
};
```

**API Route:** `DELETE /clientes/:id`

## Best Practices

<Accordion title="Input Validation">
  Always validate user input before submitting:
  
  ```javascript
  function validateClientData(nombre, correo, telefono) {
      if (!nombre || nombre.trim() === '') {
          alert('El nombre es requerido');
          return false;
      }
      
      if (!correo || !correo.includes('@')) {
          alert('Ingrese un correo válido');
          return false;
      }
      
      if (!telefono || telefono.length < 7) {
          alert('Ingrese un teléfono válido');
          return false;
      }
      
      return true;
  }
  ```
</Accordion>

<Accordion title="Error Handling">
  Implement robust error handling for better user experience:
  
  ```javascript
  async function guardarDataClienteWithErrorHandling() {
      const dataCliente = {
          nombre: document.getElementById("clientName").value,
          correo: document.getElementById("clientEmail").value,
          telefono: document.getElementById("clientPhone").value
      };
      
      // Validate before sending
      if (!validateClientData(dataCliente.nombre, dataCliente.correo, dataCliente.telefono)) {
          return;
      }

      try {
          const response = await fetch(URL, {
              method: "POST",
              headers: { 'Content-Type': 'application/json' },
              body: JSON.stringify(dataCliente)
          });

          if (!response.ok) {
              const error = await response.json();
              throw new Error(error.message || 'Error al guardar cliente');
          }

          alert('Cliente guardado exitosamente');
          loadDataCliente();
          
          // Clear form
          document.getElementById("clientName").value = '';
          document.getElementById("clientEmail").value = '';
          document.getElementById("clientPhone").value = '';

      } catch (error) {
          console.error('Error:', error);
          alert(`Error: ${error.message}`);
      }
  }
  ```
</Accordion>
